-MainLayout.jsx:

import React, { useEffect, useState } from "react";
import Topbar from "../components/Topbar";
import Sidebar from "../components/Sidebar";
import Hero from "../components/Hero";

function MainLayout() {

    // const [cosaC’èDentro, funzionePerCambiarlo] = useState(valoreIniziale);
    const [sidebarAperta, setSidebarAperta] = useState(true);

    // Carica il valore iniziale dal localStorage
    const [temaChiaro, setTemaChiaro] = useState(() => {
        const salvato = localStorage.getItem("temaChiaro");
        return salvato === null ? true : salvato === "true";
    });

    // Quando cambia temaChiaro, aggiorna il localStorage e il body
    useEffect(() => {
        localStorage.setItem("temaChiaro", temaChiaro);
        document.body.classList.remove("temaScuro", "temaChiaro");
        document.body.classList.add(temaChiaro ? "temaChiaro" : "temaScuro");
    }, [temaChiaro]);

    return (
        <>
            {/* TOPBAR */}
            <Topbar
                toggleSidebar={() => setSidebarAperta(!sidebarAperta)}
                toggleTema={() => setTemaChiaro(!temaChiaro)}
            />

            {/* SIDEBAR */}
            {sidebarAperta && <Sidebar />}      {/* condizione && cosaFareSeVera */}

            <Hero/>

        </>
    )

}

export default MainLayout

-Sidebar.jsx:

import React, { useState } from "react";
import { Button, Offcanvas, Nav } from 'react-bootstrap';
import { List } from 'react-bootstrap-icons';

function Sidebar() {
    // Stato per controllare la visibilità della sidebar su schermi piccoli
    const [show, setShow] = useState(false);

    const handleClose = () => setShow(false);
    const handleShow = () => setShow(true);

    // Contenuto della sidebar riutilizzabile
    const sidebarContent = (
        <Nav className="flex-column p-3 gap-3">
            {/* Home */}
            <Button href="/home" className="btn-secondary" active>Home</Button>
            {/* Anagrafica Clienti */}
            <Button href="/anagraficaClienti" className="btn-secondary">Anagrafica Clienti</Button>
            {/* Anagrafica Prodotti */}
            <Button href="/anagraficaProdotti" className="btn-secondary">Anagrafica Prodotti</Button>
            {/* Fatture */}
            <Button href="/fatture" className="btn-secondary">Fatture</Button>
        </Nav>
    );

    return (
        <>

            {/* Btn hamburger */}
            <Button variant="outline-secondary" className="d-lg-none m-2" onClick={handleShow}>
                <List size={24} />
            </Button>

            {/* Componente Offcanvas */}
            <Offcanvas show={show} onHide={handleClose} responsive="lg">
                <Offcanvas.Header closeButton>
                    <Offcanvas.Title>Menù</Offcanvas.Title>
                </Offcanvas.Header>
                <Offcanvas.Body className="p-0">
                    {sidebarContent}
                </Offcanvas.Body>
            </Offcanvas>
        </>
    );
}

export default Sidebar;
